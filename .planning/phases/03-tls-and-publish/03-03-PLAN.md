**IMPORTANT:** Integration tests in the `tests/` directory can only access the crate's PUBLIC API. They import `pop3::Pop3Client` etc. They CANNOT use `build_test_client` or `Transport::mock` (those are `pub(crate)`). Instead, these tests need to create mock TCP streams and exercise the connect flow.

**HOWEVER:** `Pop3Client::connect()` calls `TcpStream::connect()` which requires a real server. To test via the public API without a real server, we have two options:
1. Spawn a `tokio::net::TcpListener` that acts as a mock POP3 server
2. Keep the integration tests as inline `#[cfg(test)]` in client.rs using the mock infrastructure

Per the Phase 2 decision (02-04): "existing tokio_test::io::Builder tests satisfy the integration test criterion — they exercise the full client->transport->mock I/O path for all commands." So integration tests using the mock infrastructure in client.rs are acceptable.

**Approach:** Add integration-style tests in `src/client.rs` that test MULTI-COMMAND FLOWS (not just single commands). These simulate a realistic server conversation.

```rust
#[tokio::test]
async fn capa_then_login_then_top_flow() {
    let mock = Builder::new()
        // CAPA before login
        .write(b"CAPA\r\n")
        .read(b"+OK\r\nTOP\r\nUIDL\r\nSASL PLAIN\r\nSTLS\r\n.\r\n")
        // Login
        .write(b"USER bob\r\n")
        .read(b"+OK\r\n")
        .write(b"PASS pass123\r\n")
        .read(b"+OK welcome\r\n")
        // TOP
        .write(b"TOP 1 3\r\n")
        .read(b"+OK\r\nFrom: sender@example.com\r\nSubject: Test\r\n\r\nLine 1\r\nLine 2\r\nLine 3\r\n.\r\n")
        // QUIT
        .write(b"QUIT\r\n")
        .read(b"+OK bye\r\n")
        .build();

    let mut client = build_test_client(mock);

    // CAPA (pre-auth, per RFC 2449)
    let caps = client.capa().await.unwrap();
    assert_eq!(caps.len(), 4);
    assert!(caps.iter().any(|c| c.name == "TOP"));
    assert!(caps.iter().any(|c| c.name == "STLS"));

    // Login
    client.login("bob", "pass123").await.unwrap();

    // TOP 1 3 (headers + 3 lines)
    let msg = client.top(1, 3).await.unwrap();
    assert!(msg.data.contains("From: sender@example.com"));
    assert!(msg.data.contains("Subject: Test"));
    assert!(msg.data.contains("Line 1"));
    assert!(msg.data.contains("Line 3"));

    // Quit
    client.quit().await.unwrap();
}
```

```rust
        .read(b"+OK\r\n")

        // RSET (undelete)
        .write(b"RSET\r\n")
        .read(b"+OK\r\n")
        // NOOP
        .write(b"NOOP\r\n")
        .read(b"+OK\r\n")
        // QUIT
        .write(b"QUIT\r\n")
        .read(b"+OK\r\n")

    assert_eq!(uids[1].unique_id, "msg-bbb");

    client.dele(2).await.unwrap();

    // RSET (unmark deletion)
    client.rset().await.unwrap();

    // NOOP (keepalive)
    client.noop().await.unwrap();

    // Quit
```

```rust
//! Integration tests exercising the public Pop3Client API over real TCP.

use pop3::{Pop3Client, SessionState};
use tokio::io::{AsyncReadExt, AsyncWriteExt};
use tokio::net::TcpListener;

/// Spawn a mock POP3 server on a random port. Returns the address.
async fn spawn_mock_server(conversation: Vec<(&'static str, &'static str)>) -> String {
    let listener = TcpListener::bind("127.0.0.1:0").await.unwrap();
    let addr = listener.local_addr().unwrap();
    let addr_str = addr.to_string();

    tokio::spawn(async move {
        let (mut socket, _) = listener.accept().await.unwrap();

        // Send greeting
        socket.write_all(b"+OK Mock POP3 server ready\r\n").await.unwrap();

        for (expected_cmd, response) in conversation {
            let mut buf = vec![0u8; 1024];
            let n = socket.read(&mut buf).await.unwrap();
            let received = String::from_utf8_lossy(&buf[..n]);
            assert_eq!(received.trim(), expected_cmd, "unexpected command");
            socket.write_all(response.as_bytes()).await.unwrap();
        }
    });

    addr_str
}

#[tokio::test]
async fn public_api_connect_login_stat_quit() {
    let addr = spawn_mock_server(vec![
        ("USER testuser", "+OK\r\n"),
        ("PASS testpass", "+OK logged in\r\n"),
        ("STAT", "+OK 3 15000\r\n"),
        ("QUIT", "+OK goodbye\r\n"),
    ]).await;

    let mut client = Pop3Client::connect(
        addr.as_str(),
        std::time::Duration::from_secs(5),
    ).await.unwrap();

    assert_eq!(client.state(), SessionState::Connected);
    assert!(!client.is_encrypted());

    client.login("testuser", "testpass").await.unwrap();
    assert_eq!(client.state(), SessionState::Authenticated);

    let stat = client.stat().await.unwrap();
    assert_eq!(stat.message_count, 3);
    assert_eq!(stat.mailbox_size, 15000);

    client.quit().await.unwrap();
}

#[tokio::test]
async fn public_api_capa_and_top() {
    let addr = spawn_mock_server(vec![
        ("CAPA", "+OK\r\nTOP\r\nUIDL\r\nSASL PLAIN\r\n.\r\n"),
        ("USER user", "+OK\r\n"),
        ("PASS pass", "+OK\r\n"),
        ("TOP 1 5", "+OK\r\nSubject: Test\r\n\r\nBody line\r\n.\r\n"),
        ("QUIT", "+OK\r\n"),
    ]).await;

    let mut client = Pop3Client::connect(
        addr.as_str(),
        std::time::Duration::from_secs(5),
    ).await.unwrap();

    assert!(caps.iter().any(|c| c.name == "TOP"));

    client.login("user", "pass").await.unwrap();

    let msg = client.top(1, 5).await.unwrap();
    assert!(msg.data.contains("Subject: Test"));

    client.quit().await.unwrap();
}
```
