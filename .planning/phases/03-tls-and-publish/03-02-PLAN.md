**Part A — OpenSSL connect_tls:**

Add the openssl-tls feature-gated version of `connect_tls`. This method has the SAME signature as the rustls version — the `#[cfg]` gates ensure only one compiles at a time.

```rust
#[cfg(feature = "openssl-tls")]
pub(crate) async fn connect_tls(
    addr: impl tokio::net::ToSocketAddrs,
    hostname: &str,
    timeout: Duration,
) -> Result<Self> {
    use openssl::ssl::{SslConnector, SslMethod};
    use tokio_openssl::SslStream;
    use std::pin::Pin;

    let connector = SslConnector::builder(SslMethod::tls())
        .map_err(|e| Pop3Error::Tls(e.to_string()))?
        .build();

    let ssl = connector.configure()
        .map_err(|e| Pop3Error::Tls(e.to_string()))?
        .into_ssl(hostname)
        .map_err(|e| Pop3Error::Tls(e.to_string()))?;

    let tcp_stream = TcpStream::connect(addr).await?;
    let mut tls_stream = SslStream::new(ssl, tcp_stream)
        .map_err(|e| Pop3Error::Tls(e.to_string()))?;

    Pin::new(&mut tls_stream).connect().await
        .map_err(|e| Pop3Error::Tls(e.to_string()))?;

    let inner = InnerStream::OpensslTls(tls_stream);
    let (read_half, write_half) = io::split(inner);
    Ok(Transport {
        reader: BufReader::new(read_half),
        writer: write_half,
        timeout,
        encrypted: true,
    })
}
```

Note: `into_ssl(hostname)` enables SNI and hostname verification automatically. OpenSSL reads the system CA bundle by default on Linux/macOS. No explicit trust store loading needed.

**Part B — STARTTLS upgrade_in_place:**

Implement `upgrade_in_place(&mut self, hostname: &str) -> Result<()>` which upgrades a plain TCP connection to TLS on the SAME socket. This is the critical method for STARTTLS.

The approach uses `tokio::io::ReadHalf::unsplit()` to recover the `InnerStream`, extracts the `TcpStream` from the `Plain` variant, performs a TLS handshake, then re-splits and rebuilds the Transport fields.

```rust
#[cfg(any(feature = "rustls-tls", feature = "openssl-tls"))]
pub(crate) async fn upgrade_in_place(&mut self, hostname: &str) -> Result<()> {
    // TLS-06: Verify BufReader buffer is empty before upgrade.
    // Any pending bytes would be lost during the reader replacement.
    let pending = self.reader.buffer().len();
    if pending > 0 {
        return Err(Pop3Error::Io(io::Error::new(
            io::ErrorKind::InvalidData,
            format!("unexpected {} bytes in buffer before TLS upgrade", pending),
        )));
    }

    if self.encrypted {
        return Err(Pop3Error::Tls("connection is already encrypted".into()));
    }

    // Step 1: Take reader and writer out of self using mem::replace with
    // temporary placeholders. We use io::empty()/io::sink() as dummies.
    // These are immediately replaced so they never perform actual I/O.
    let dummy_inner = InnerStream::dummy(); // See below
    let (dummy_read, dummy_write) = io::split(dummy_inner);
    let old_reader = std::mem::replace(
        &mut self.reader,
        BufReader::new(dummy_read),
    );
    let old_writer = std::mem::replace(&mut self.writer, dummy_write);

    // Step 2: Recover the InnerStream from the split halves.
    // into_inner() is safe because we verified the buffer is empty above.
    let read_half = old_reader.into_inner();
    let inner_stream = read_half.unsplit(old_writer);

    // Step 3: Extract the TcpStream from the Plain variant.
    let tcp_stream = match inner_stream {
        InnerStream::Plain(tcp) => tcp,
        _ => {
            return Err(Pop3Error::Tls(
                "upgrade_in_place requires a plain TCP connection".into()
            ));
        }
    };

    // Step 4: Perform TLS handshake on the existing TCP stream.
    let tls_inner = Self::tls_handshake(tcp_stream, hostname).await?;

    // Step 5: Split the new TLS stream and rebuild self.
    let (read_half, write_half) = io::split(tls_inner);
    self.reader = BufReader::new(read_half);
    self.writer = write_half;
    self.encrypted = true;

    Ok(())
}
```

**Helper: tls_handshake** — performs the TLS handshake on a TcpStream, returns the appropriate InnerStream variant:

```rust
#[cfg(feature = "rustls-tls")]
async fn tls_handshake(tcp_stream: TcpStream, hostname: &str) -> Result<InnerStream> {
    use std::sync::Arc;
    use tokio_rustls::TlsConnector;
    use tokio_rustls::rustls::{ClientConfig, RootCertStore};

    let server_name = tokio_rustls::rustls::pki_types::ServerName::try_from(hostname.to_owned())
        .map_err(|e| Pop3Error::InvalidDnsName(e.to_string()))?;

    let native_certs = rustls_native_certs::load_native_certs();
    let mut root_store = RootCertStore::empty();
    for cert in native_certs.certs {
        root_store.add(cert).map_err(|e| Pop3Error::Tls(e.to_string()))?;
    }

    let config = ClientConfig::builder()
        .with_root_certificates(root_store)
        .with_no_client_auth();
    let connector = TlsConnector::from(Arc::new(config));

    let tls_stream = connector.connect(server_name, tcp_stream).await
        .map_err(|e| Pop3Error::Tls(e.to_string()))?;

    Ok(InnerStream::RustlsTls(tls_stream))
}

#[cfg(feature = "openssl-tls")]
async fn tls_handshake(tcp_stream: TcpStream, hostname: &str) -> Result<InnerStream> {
    use openssl::ssl::{SslConnector, SslMethod};
    use tokio_openssl::SslStream;
    use std::pin::Pin;

    let connector = SslConnector::builder(SslMethod::tls())
        .map_err(|e| Pop3Error::Tls(e.to_string()))?
        .build();

    let ssl = connector.configure()
        .map_err(|e| Pop3Error::Tls(e.to_string()))?
        .into_ssl(hostname)
        .map_err(|e| Pop3Error::Tls(e.to_string()))?;

    let mut tls_stream = SslStream::new(ssl, tcp_stream)
        .map_err(|e| Pop3Error::Tls(e.to_string()))?;

    Pin::new(&mut tls_stream).connect().await
        .map_err(|e| Pop3Error::Tls(e.to_string()))?;

    Ok(InnerStream::OpensslTls(tls_stream))
}
```

```rust
/// Upgrade the connection to TLS via the STLS command (POP3 STARTTLS).
///
/// Must be called before authentication — STLS is only valid in the
/// AUTHORIZATION state per RFC 2595. After a successful upgrade,
/// [`is_encrypted()`](Self::is_encrypted) returns `true`.
///
/// The `hostname` parameter is used for TLS server name verification (SNI).
#[cfg(any(feature = "rustls-tls", feature = "openssl-tls"))]
pub async fn stls(&mut self, hostname: &str) -> Result<()> {
    // RFC 2595: STLS is only valid in AUTHORIZATION state (before login)
    if self.state == SessionState::Authenticated {
        return Err(Pop3Error::ServerError(
            "STLS not allowed after authentication (RFC 2595)".into(),
        ));
    }
    if self.is_encrypted() {
        return Err(Pop3Error::ServerError(
            "connection is already encrypted".into(),
        ));
    }

    // Send STLS command and verify server accepts
    self.send_and_check("STLS").await?;

    // Upgrade the underlying transport to TLS
    self.transport.upgrade_in_place(hostname).await?;

    Ok(())
}
```
